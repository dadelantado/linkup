Agenda de Contactos

El siguiente código trata sobre un proyecto de diferentes archivos escritos en PHP que construyen un proyecto para una agenda de Contactos.

conexion.php

<?php
$host = "localhost";
$user = "root";
$password = "";
$dbname = "agenda_db";

try{
    $conn = new PDO("mysql:host =$host;dbname=$dbname;charset=utf8", $user, $password);

    $conn->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
    echo "Conexion exitosa <br>";

}catch (PDOException $e){
    die("Conexion fallida " . $e->getMessage());
}

?>

insertar.php

<?php 

function añadirContacto($Nombre, $telefono_Oficina, $telefono_Casa, $Celular, $Empresa, $Cargo, $Fecha_de_Cumpleanos, $conn) {
    // Consulta SQL con parámetros nombrados
    $sql = "INSERT INTO contactos (Nombre, telefono_Oficina, telefono_Casa, Celular, Empresa, Cargo, Fecha_de_Cumpleanos) 
            VALUES (:Nombre, :telefono_Oficina, :telefono_Casa, :Celular, :Empresa, :Cargo, :Fecha_de_Cumpleanos)";

    try {
        $stmt = $conn->prepare($sql);

        // Bind de parámetros usando los nombres de parámetros
        $stmt->bindParam(':Nombre', $Nombre, PDO::PARAM_STR);
        $stmt->bindParam(':telefono_Oficina', $telefono_Oficina, PDO::PARAM_STR);
        $stmt->bindParam(':telefono_Casa', $telefono_Casa, PDO::PARAM_STR);
        $stmt->bindParam(':Celular', $Celular, PDO::PARAM_STR);
        $stmt->bindParam(':Empresa', $Empresa, PDO::PARAM_STR);
        $stmt->bindParam(':Cargo', $Cargo, PDO::PARAM_STR);
        $stmt->bindParam(':Fecha_de_Cumpleanos', $Fecha_de_Cumpleanos, PDO::PARAM_STR);

        // Ejecuta la consulta y muestra el resultado
        if ($stmt->execute()) {
            echo "Datos insertados correctamente";
        } else {
            echo "Error al insertar datos";
        }
    } catch (PDOException $e) {
        echo "Error en la preparación de la consulta: " . $e->getMessage();
    }
}
?>
